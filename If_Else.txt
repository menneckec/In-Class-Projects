#include "stdafx.h"
#include<iostream>
#include<string>
using namespace std;

int main()
{
	//Declare variables and constants//
	int HIGH_SCORE = 95;

	string fullName;

	double testScoreOne;
	double testScoreTwo;
	double testScoreThree;

	double testScoreAverage;
	char letterGrade;

	//Request information//
	cout << "Please enter your full name." << endl;
	getline(cin, fullName);

	if (fullName == "NO-NAME") {
		cout << "Invalid name." << endl;
		return 0;
	}
	else {
		cout << "Please enter your last three test scores: " << endl;
		cout << "Test 1: ";
		cin >> testScoreOne;
		cout << "Test 2: ";
		cin >> testScoreTwo;
		cout << "Test 3: ";
		cin >> testScoreThree;
	}

	//Calculate average test scores//
	testScoreAverage = (testScoreOne + testScoreTwo + testScoreThree) / 3;

	//Assign grades based on test scores//
	if (testScoreAverage >= HIGH_SCORE) {
		letterGrade = 'A';
	}
	else letterGrade = 'B';

	//Display results//
	if (letterGrade == 'A') {
		cout << "Average score: " << testScoreAverage << endl;
		cout << "Letter grade: " << letterGrade << endl;
		cout << "Congratulations!" << endl;
	}
	else  {
		cout << "Average score: " << testScoreAverage << endl;
	}

	system("pause");
    return 0;
}
