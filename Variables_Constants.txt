#include "stdafx.h"
#include <iostream>
using namespace std;

int main()
{
	//Declare constants//
	const double TAX_RATE = 0.0675;
	const double TIP_RATE = .2;

	//Declare variables//
	double mealCost = 88.67;
	double mealTax;
	double mealCostWithTax;
	double mealTip;
	double mealTotal;

	//Calculate tax, tip, and total//
	mealTax = mealCost*TAX_RATE;
	mealCostWithTax = mealCost + mealTax;
	mealTip = mealCostWithTax*TIP_RATE;
	mealTotal = mealCostWithTax + mealTip;

	//Display meal price, tax, tip, and total//
	cout << "The original cost of the meal is $" << mealCost << endl;
	cout << "The tax on the meal amounts to $" << mealTax << endl;
	cout << "The cost of the meal after tax and before tipping is $" << mealCostWithTax << endl;
	cout << "The amount of a 20% tip is $" << mealTip << endl;
	cout << "The total cost of the meal amounts to $" << mealTotal << endl;

	//Pauses the system to view results, then return to Visual Studio//
	system("pause");
    return 0;
}
