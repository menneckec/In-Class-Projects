#include "stdafx.h"
#include <iostream>
#include <iomanip>
using namespace std;


int main()
{
	//Declare variables//
	int userSelection;

	double circleRadius;
	double PI = 3.14159;
	double circleArea;

	double rectangleLength;
	double rectangleWidth;
	double rectangleArea;

	double triangleBase;
	double triangleHeight;
	double triangleArea;

	cout << fixed << setprecision(2) << showpoint;

	//Display menu//
	cout << "Geometry Calculator" << endl;
	cout << setw(5) << "1. " << "Calculate the area of a circle." << endl;
	cout << setw(5) << "2. " << "Calculate the area of a rectangle." << endl;
	cout << setw(5) << "3. " << "Calculate the area of a triangle." << endl;
	cout << setw(5) << "4. " << "Quit.\n" << endl;

	//Request input//
	cout << "Enter your choice (1-4)" << endl;
	cin >> userSelection;

	switch (userSelection) {
	case 1: 
		cout << "Finding the area of a circle. Please enter data below:" << endl;
		cout << "Radius (cm): ";
		cin >> circleRadius;

		if (circleRadius < 0) {
			cout << "Invalid data. Please enter a positive number."<<endl;
			system("pause");
			return 0;
		}
		else

		//Perform calculations//
		circleArea = PI*(circleRadius*circleRadius);

		//Display results//
		cout << "You have entered the following data:" << endl << "Radius: " << circleRadius << " cm" << endl;
		cout << "Using this data, the area of your circle is: " << circleArea << " cm" << endl;

		break;
	case 2: 
		cout << "Finding the area of a rectangle. Please enter data below:" << endl;
		cout << "Length (cm): ";
		cin >> rectangleLength;

		if (rectangleLength < 0) {
			cout << "Invalid data. Please enter a positive number."<<endl;
			system("pause");
			return 0;
		}
		else

		cout << "Width (cm): ";
		cin >> rectangleWidth;

		if (rectangleWidth < 0) {
			cout << "Invalid data. Please enter a positive number."<<endl;
			system("pause");
			return 0;
		}
		else

		//Perform calculations//
		rectangleArea = rectangleLength*rectangleWidth;

		//Display results//
		cout << "You have entered the following data:" << endl << "Length: " << rectangleLength << " cm" << endl << "Width: " << rectangleWidth << " cm" << endl;
		cout << "Using this data, the area of your rectangle is: " << rectangleArea << " cm" << endl;

		break;
	case 3: 
		cout << "Finding the area of a triangle. Please enter data below:" << endl;
		cout << "Length of base (cm): ";
		cin >> triangleBase;

		if (triangleBase < 0) {
			cout << "Invalid data. Please enter a positive number."<<endl;
			system("pause");
			return 0;
		}
		else

		cout << "Height(cm): ";
		cin >> triangleHeight;

		if (triangleHeight < 0) {
			cout << "Invalid data. Please enter a positive number."<<endl;
			system("pause");
			return 0;
		}
		else

		//Perform calculations//
		triangleArea = triangleBase*triangleHeight / 2;

		//Display results//
		cout << "You have entered the following data:" << endl << "Length of base: " << triangleBase << " cm" << endl << "Height: " << triangleHeight << " cm" << endl;
		cout << "Using this data, the area of your triangle is: " << triangleArea << " cm" << endl;
		break;
	case 4: 
		cout << "You did not choose any shape that we can calculate the area. Quitting the program." << endl;
		system("pause");
		return 0;
	default: cout << "Invalid data. Quitting the program." << endl;
		system("pause");
		return 0;
	}

	system("pause");
    return 0;
}
